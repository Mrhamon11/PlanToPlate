# users/models.py
from django.db import models
from django.contrib.auth.models import AbstractBaseUser, BaseUserManager
from django.utils import timezone


class CustomUser(AbstractBaseUser):
    """
    Custom user model with email-based authentication.
    
    Fields:
    - Email: Login credential and unique identifier
    - Username: Optional display name (derived from email or admin-set)
    - First/Last Name: Optional profile fields
    
    Multi-tenancy:
    - Every domain object references this model via created_by FK
    - Shared objects reference via Permission model's granted_to FK
    
    Use Case: Replace Django's built-in User model for custom auth logic
    """
    
    # Core authentication fields
    email = models.EmailField(
        unique=True,
        help_text="Email address for login"
    )
    
    username = models.CharField(
        max_length=150,
        blank=True,
        null=True,
        help_text="Optional display name (can be empty)"
    )
    
    # Email address verification
    is_email_verified = models.BooleanField(
        default=False,
        help_text="Has this user verified their email?"
    )
    
    # Account status
    is_active = models.BooleanField(
        default=True,
        help_text="Is this account currently active?"
    )
    
    # Timestamps
    date_joined = models.DateTimeField(
        default=timezone.now
    )
    
    last_login = models.DateTimeField(
        null=True,
        blank=True
    )
    
    # Admin fields
    is_staff = models.BooleanField(
        default=False,
        help_text="Is this user a staff member (admin access)?"
    )
    
    class Meta:
        verbose_name = 'User'
        verbose_name_plural = 'Users'
        ordering = ['email']  # Email-based ordering
    
    def __str__(self):
        return self.email


class CustomUserManager(BaseUserManager):
    """Custom manager to create users from email addresses."""
    
    def create_user(self, email, password=None, **extra_fields):
        if not email:
            raise ValueError('Users must have an email address')
        
        # Normalize email address (remove whitespace, lowercase)
        email = email.strip().lower()
        
        user = self.model(email=email, **extra_fields)
        user.set_password(password)
        user.save(using=self._db)
        
        return user
    
    def create_superuser(self, email, password=None, **extra_fields):
        extra_fields.setdefault('is_staff', True)
        extra_fields.setdefault('is_superuser', True)
        extra_fields.setdefault('is_active', True)
        
        if extra_fields.get('email_verified') is None:
            extra_fields['is_email_verified'] = True
        
        return self.create_user(email, password, **extra_fields)


class Permission(models.Model):
    """
    Cross-app permission model for multi-tenancy sharing.
    
    This model exists in the `users/` app as it's shared across all domain apps:
    - ingredients: Ingredient objects can be shared with specific users
    - recipes: Recipe objects can be shared via Permission links
    - dishes, lists, recipebooks: Same pattern
    
    Use Cases:
    1. User A creates a recipe (owned_by=User A)
    2. User A wants to share this recipe with User B (read-only)
    3. Create Permission object linking them via created_by/granted_to
    
    Structure:
    - owner: User who owns the object being shared (or original creator)
    - granted_to: User who gets access permissions (can be None for self-access)
    - object_type: Model type name string (ingredient, recipe, etc.)
    - object_id: Primary key of the target object
    - read_only: False = can edit/delete; True = view-only access
    
    Multi-tenancy Rules:
    1. Default behavior: Queryset must filter by owner=user unless user owns it OR has Permission record granting access
    2. Public objects: Objects with is_public=True bypass permission filtering (optional flag on domain models)
    3. Explicit sharing: Permission model provides granular control beyond public/private flags
    
    Use Case Example:
        # User A shares a recipe with User B
        from users.models import Permission
        
        permission = Permission.objects.create(
            object_type='recipe',
            object_id=42,           # The Recipe's primary key
            owner=User_A,           # Original creator
            granted_to=User_B,      # Gets read access
            read_only=True          # Can view but not edit/delete
        )
        
        # Query recipes for User B:
        # Must check both owned AND permission records
        PermissionQuerySet(recipe_qs).for_user(User_B)
    """
    
    # Reference to the object being shared (string reference, not FK)
    object_type = models.CharField(
        max_length=50,
        help_text="Model type name (ingredient, recipe, dish, etc.)"
    )
    
    object_id = models.BigIntegerField()
    
    # Owner of the original object (or creator who owns it)
    owner = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        help_text="User who created/owns this object"
    )
    
    # User granted permission to access this object
    granted_to = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        related_name='permission_access',  # All permissions for this user
        blank=True,
        null=True
    )
    
    # Read-only or full access
    read_only = models.BooleanField(
        default=True,
        help_text="True = view-only; False = can edit/delete"
    )
    
    # Metadata (optional)
    shared_reason = models.TextField(
        blank=True,
        help_text="Why this object was shared (optional)"
    )
    
    created_at = models.DateTimeField(
        auto_now_add=True
    )
    
    class Meta:
        verbose_name = 'Permission'
        verbose_name_plural = 'Permissions'
        ordering = ['-created_at']
        
        # Index for efficient querying by granted user and object type
        indexes = [
            models.Index(fields=['granted_to', 'object_type']),
        ]
    
    def __str__(self):
        return f"{self.owner.username} -> {self.granted_to.username}: {self.object_type} #{self.object_id}"


# API Contract for Authentication

### 2.2 users/admin.py - Admin Registration

```python
# users/admin.py
from django.contrib import admin
from django.contrib.auth.admin import UserAdmin as DefaultUserAdmin
from .models import CustomUser, Permission


@admin.register(CustomUser)
class CustomUserAdmin(DefaultUserAdmin):
    """Admin interface for user management."""
    
    fieldsets = DefaultUserAdmin.fieldsets + (
        ('Multi-tenancy', {
            'fields': ('is_staff', 'is_active', 'is_email_verified')
        }),
        ('Profile', {
            'fields': ('username', 'last_login')
        }),
    )
    
    list_display = ['email', 'username', 'is_staff', 'is_active']
    search_fields = ['email', 'username']
    ordering = ['email']


@admin.register(Permission)
class PermissionAdmin(admin.ModelAdmin):
    """Admin interface for cross-app permissions."""
    
    list_display = [
        'object_type', 
        'owner_username', 
        'granted_to_username',
        'access_level',
        'created_at'
    ]
    
    search_fields = [
        'owner__email',
        'granted_to__email',
        'object_type'
    ]
    
    list_filter = ['object_type', 'read_only']
    
    raw_id_fields = ['owner', 'granted_to']
    
    fieldsets = DefaultUserAdmin.fieldsets + (
        ('Object Reference', {
            'fields': ('object_type', 'object_id')
        }),
        ('Shared With', {
            'fields': ('granted_to', 'read_only')
        }),
        ('Metadata', {
            'fields': ('shared_reason', 'created_at')
        }),
    )
    
    def owner_username(self, obj):
        return obj.owner.email if hasattr(obj.owner, 'email') else str(obj.owner)
    owner_username.short_description = 'Owner Email'
    
    def granted_to_username(self, obj):
        return obj.granted_to.email if hasattr(obj.granted_to, 'email') else str(obj.granted_to)
    granted_to_username.short_description = 'Granted To Email'
    
    def access_level(self, obj):
        return 'Read-Only' if obj.read_only else 'Full Access'
    access_level.short_description = 'Access Level'
```
