```python
# dishes/models.py
from django.db import models
from coremodels.models import BaseModel


class Dish(BaseModel):
    """
    A dish is a collection of recipes that together form a complete meal.
    
    Use Cases:
    - Combine multiple recipes into one meal (e.g., "Dinner Special" = Main + Side + Dessert)
    - Group for weekly meal planning
    - Dish can be added to multiple lists (shopping list, menu plan, etc.)
    
    Structure:
    - Dish has many Recipes (M2M or FK with intermediate table)
    - Each recipe's ingredients are already stored in recipes.app
    - No duplicate ingredient storage here
    
    Multi-tenancy:
    - Default: user sees only their owned dishes
    - Optional `is_public` for shared meals
    - Recipe visibility determined by each recipe's own permissions
    
    API: GET /internal/dishes/ lists all dishes visible to current user
          POST /internal/dishes/ creates new dish (owned by authenticated user)
    
    Example Structure:
        Dish("Dinner Special") = 
            - Main Course: Pasta Recipe
            - Side: Salad Recipe  
            - Dessert: Fruit Bowl Recipe
    """
    
    # Core fields
    name = models.CharField(
        max_length=200,
        help_text="Name of the dish (e.g., 'Dinner Special', 'Lunch Combo')"
    )
    
    description = models.TextField(
        blank=True,
        help_text="Description of the meal components"
    )
    
    # Meal type classification
    meal_type = models.CharField(
        max_length=50,
        choices=[
            ('appetizer', 'Appetizer'),
            ('main_course', 'Main Course'),
            ('side_dish', 'Side Dish'),
            ('dessert', 'Dessert'),
            ('soup', 'Soup'),
            ('salad', 'Salad'),
        ],
        help_text="Type of dish for menu planning"
    )
    
    # Multi-tenancy flags
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this dish (read-only). Owners keep full CRUD."
    )
    
    # Relationships with recipes
    # Using FK with intermediate table for ordering control
    dishes_recipes = models.ManyToManyField(
        'recipes.Recipe',
        through='DishRecipe',
        related_name='dish_groups'  # All dishes this recipe is part of
    )
    
    class Meta:
        verbose_name = 'Dish'
        verbose_name_plural = 'Dishes'
        ordering = ['-updated_at']
    
    def __str__(self):
        return self.name
    
    @property
    def flat_ingredients(self):
        """
        Get flat list of all ingredients from all recipes in this dish.
        Returns: [{'name': 'Pasta', 'quantity': '500g', 'unit': 'g'}, ...]
        """
        from recipes.models import RecipeIngredient, Recipe
        from ingredients.models import Ingredient
        
        ingredient_list = []
        
        for recipe in self.dishes_recipes.all():
            # Get recipe's direct ingredients
            for ri in RecipeIngredient.objects.filter(recipe=recipe):
                # Add simple ingredient or nested recipe name
                if ri.ingredient_ref:
                    ingredient_list.append({
                        'name': ri.ingredient_ref.name,
                        'quantity': f"{float(ri.quantity_string)} {ri.unit}",
                        'unit': ri.unit,
                    })
                elif ri.nested_recipe:
                    # Get nested recipe's ingredients recursively (simplified)
                    for sub_ri in RecipeIngredient.objects.filter(recipe=ri.nested_recipe):
                        if sub_ri.ingredient_ref:
                            ingredient_list.append({
                                'name': sub_ri.ingredient_ref.name,
                                'quantity': f"{float(sub_ri.quantity_string)} {sub_ri.unit}",
                                'unit': sub_ri.unit,
                            })
        
        return ingredient_list

        total = 0

        return total


# Intermediate table for ordering control
class DishRecipe(models.Model):
    """
    Intermediate M2M table allowing ordered assignment of recipes to dishes.
    
    Use Case: Ensure main course comes before dessert in meal plan
    """
    dish = models.ForeignKey(
        'dishes.Dish',
        on_delete=models.CASCADE,
        related_name='dish_recipe_assignments'
    )
    
    recipe = models.ForeignKey(
        'recipes.Recipe',
        on_delete=models.CASCADE,
        related_name='assigned_to_dishes'
    )
    
    order = models.PositiveIntegerField(
        help_text="Order in which this dish component appears"
    )
    
    class Meta:
        ordering = ['order']
        unique_together = ('dish', 'recipe')  # Prevent duplicate assignments
    
    def __str__(self):
        return f"{self.dish.name} - {self.recipe.name}"
```

```python
    Create new dish owned by authenticated user.
    """
    
    def post(self, request):
        from django.contrib.auth.models import AnonymousUser

            return Response({'error': 'Authentication required'}, status=401)

        dish = Dish.objects.create(
            name=data.get('name'),
            description=data.get('description', ''),
            meal_type=data.get('meal_type', 'main_course'),

            favorite=data.get('favorite', False),
        )
        
        return Response({
            'id': dish.id,
            'name': dish.name,
        }, status=201)
```

### 5.5 dishes/serializers.py - DRF Serializer with Permission Filtering

```python
# dishes/serializers.py
from rest_framework import serializers
from django.contrib.auth.models import AnonymousUser
from .models import Dish


class DishSerializer(serializers.ModelSerializer):
    """
    DRF serializer for external API.
    Applies multi-tenancy filtering based on current user context.
    """
    
    class Meta:
        model = Dish
        fields = [
            'id', 'name', 'description', 'meal_type',
            'rating', 'favorite', 'notes', 'created_at', 'updated_at'
        ]
    
    def get_queryset(self):
        """
        Apply multi-tenancy filter based on authenticated user context.
        
        Returns dishes:
        - Owned by current user (full CRUD access)
        - OR public dishes (read-only for everyone)
        """
        queryset = Dish.objects.all()
        
        # Get current user from context
        user = self.context.get('request') and self.context['request'].user
        
        if user:
            queryset = queryset.filter(
                is_public=True  # Public dishes visible to all
            ) | queryset.filter(
                created_by=user  # Owned by authenticated user
            )
        
        return queryset
    
    def create(self, validated_data):
        """
        Set created_by to authenticated user on creation.
        """
        request = self.context.get('request')
        if request and hasattr(request, 'user') and not request.user.is_anonymous:
            validated_data['created_by'] = request.user
        
        return super().create(validated_data)
```
