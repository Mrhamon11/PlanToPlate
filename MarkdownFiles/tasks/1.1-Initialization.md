### 1.1.1 coremodels/models.py - BaseModel Mixin with FK Ownership

```python
# coremodels/models.py
from django.db import models
from django.utils import timezone
import uuid


class BaseModel(models.Model):
    """
    Base model mixin for all domain objects.
    
    Structure:
    - All models MUST inherit from this mixin OR define owner field directly
    - Ownership via ForeignKey to CustomUser (not email-based)
    - No abstract=True here - inheritance is explicit
    
    Use Case: Every recipe, ingredient, dish needs an owner user
    
    Fields:
    - id: Automatic PK (BigIntegerField by default)
    - rating: Default 0-5 star rating field for all objects
    - favorite: Boolean flag for user's favorites list
    - notes: Optional text notes about this object
    - created_by: FK to CustomUser (ownership reference)
    
    Multi-tenancy Pattern:
    - Objects inherit from BaseModel with created_by field
    - Queryset must filter by: created_by=user OR has Permission record granting access
    - Exceptions: is_public=True flag bypasses filtering (optional)
    
    Example Usage:
        # recipes/models.py
        class Recipe(models.Model):
            class Meta:
                abstract = False  # Actually inherit from BaseModel
            
            name = models.CharField(max_length=200)
            description = models.TextField(blank=True)
            
            # Inheritance from BaseModel gives these fields automatically:
            #   id, rating, favorite, notes, created_by
            
            # Define owner field explicitly if not inheriting from BaseModel
            owner = models.ForeignKey('users.CustomUser', on_delete=models.CASCADE, blank=True, null=True)
    """
    
    # Common metadata fields across all objects
    rating = models.FloatField(
        default=0.0,
        validators=[
            MinValueValidator(0),
            MaxValueValidator(5),
        ],
        help_text="Overall rating (0-5)"
    )
    
    favorite = models.BooleanField(
        default=False,
        help_text="Is this object marked as a favorite?"
    )
    
    notes = models.TextField(
        blank=True,
        help_text="Optional notes about this object"
    )
    
    # Ownership reference (FK to CustomUser)
    created_by = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,  # Delete objects when user is deleted
        related_name='owned_objects',  # All objects owned by this user
        blank=True,
        null=True
    )
    
    class Meta:
        abstract = True
    
    def serialize(self):
        """Return dictionary representation for API responses."""
        data = {
            'id': self.id,
            'name': getattr(self, 'name', ''),  # Not all models have name
            'rating': self.rating,
            'favorite': self.favorite,
            'notes': self.notes,
        }
        
        # Add created_by reference without full object
        if self.created_by:
            data['owner_id'] = self.created_by.id
            data['owner_email'] = self.created_by.email
        
        return data


# Optional: Shared permission queryset helper (can be in permissions.py)
from django.db.models import Q


class PermissionQuerySet(models.QuerySet):
    """
    QuerySet extension for multi-tenancy filtering.
    
    Use Case: Apply filtered queries across all domain models
    
    Example:
        from coremodels.permissions import PermissionQuerySet
        from recipes.models import Recipe
        
        qs = Recipe.objects.all()
        filtered_qs = PermissionQuerySet(qs).for_user(request.user)
    """
    
    def for_user(self, user):
        """
        Filter queryset by:
        1. Objects owned by this user (created_by=user)
        2. OR objects where user has permission to view (Permission model)
        3. OR objects marked as public
        
        Returns: Filtered QuerySet
        """
        # Get permissions for this user
        perms = Permission.objects.filter(
            Q(owner=user) |  # Objects owned by user
            Q(granted_to=user, read_only=True)  # Explicit permission to view
        )
        
        # Build filter expression: 
        # (created_by=user OR owner=user AND object_id in permissions)
        return self.filter(
            Q(created_by=user) |  # Owned by this user
            Q(created_by__permissions=perms, read_only=True)
        )
```

```python
# users/models.py (created in Task 1.2)
from django.db import models
from django.contrib.auth.models import AbstractBaseUser, BaseUserManager
from django.utils import timezone


class CustomUser(AbstractBaseUser):
    """Custom user model with email-based authentication."""
    
    email = models.EmailField(unique=True)
    username = models.CharField(max_length=150, blank=True, null=True)
    is_active = models.BooleanField(default=True)
    date_joined = models.DateTimeField(default=timezone.now)
    is_staff = models.BooleanField(default=False)
    
    objects = CustomUserManager()
    
    def __str__(self):
        return self.email


class Permission(models.Model):
    """Cross-app permission model for multi-tenancy sharing."""
    
    object_type = models.CharField(max_length=50)
    object_id = models.BigIntegerField()
    owner = models.ForeignKey('users.CustomUser', on_delete=models.CASCADE)
    granted_to = models.ForeignKey(
        'users.CustomUser', 
        on_delete=models.CASCADE,
        related_name='permission_access',
        blank=True, null=True
    )
    read_only = models.BooleanField(default=True)
    
    class Meta:
        ordering = ['-created_at']
```

```python
from django.contrib.auth.admin import UserAdmin as DefaultUserAdmin
from .models import CustomUser, Permission


@admin.register(CustomUser)
class CustomUserAdmin(DefaultUserAdmin):
    fieldsets = DefaultUserAdmin.fieldsets + (
        ('Multi-tenancy', {
            'fields': ('is_staff', 'is_active', 'is_email_verified')
        }),
    )
    
    list_display = ['email', 'username', 'is_staff']


@admin.register(Permission)
class PermissionAdmin(admin.ModelAdmin):
    list_display = ['object_type', 'owner_username', 'granted_to_username', 
                    'read_only', 'created_at']
    raw_id_fields = ['owner', 'granted_to']
```
