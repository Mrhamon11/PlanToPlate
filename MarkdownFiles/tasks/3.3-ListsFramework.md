# lists/models.py
from django.db import models
from coremodels.models import BaseModel


class ListType(models.Model):
    """
    Enumerated types of lists users can create.
    
    Use Cases:
    - Shopping Lists (containing ingredients)
    - Meal Plans (containing dishes/recipes for each day)
    - Menu Planning (weekly menu)
    - Note Lists (free-form text items)
    
    Each list type has its own default behavior and filtering rules.
    """
    
    name = models.CharField(
        max_length=100,
        unique=True,
        help_text="List type name (e.g., 'Shopping List', 'Meal Plan')"
    )
    
    slug = models.SlugField(
        unique=True,
        max_length=100,
        help_text="URL-friendly version for API endpoints"
    )
    
    description = models.TextField(blank=True)
    
    class Meta:
        verbose_name = 'List Type'
        verbose_name_plural = 'List Types'
        ordering = ['name']
    
    def __str__(self):
        return self.name


class List(BaseModel):
    """
    A user-created list that can contain multiple items of various types.
    
    Use Cases:
    - Shopping List (contains ingredients)
    - Weekly Meal Plan (contains Dishes for each day)
    - Menu Planning (contains Recipe references)
    - Notes/Notes (free-form text items)
    
    Structure:
    - List has many Items (via ListItem model)
    - Each item can reference: Ingredient, Recipe, Dish, or free-form text
    - List type determines default behavior
    
    Multi-tenancy:
    - Default: user sees only their owned lists
    - Optional `is_public` for shared lists
    - Item visibility determined by underlying model's permissions
    
    API: GET /internal/lists/ lists all lists visible to current user
          POST /internal/lists/ creates new list (owned by authenticated user)
    """
    
    # Core fields
    name = models.CharField(
        max_length=200,
        help_text="List name (e.g., 'Weekly Shopping', 'Week 1 Meal Plan')"
    )
    
    description = models.TextField(
        blank=True,
        help_text="Description of this list's purpose"
    )
    
    # List type determines behavior
    list_type = models.ForeignKey(
        ListType,
        on_delete=models.CASCADE,
        related_name='user_lists',  # All lists of this type for user
        help_text="Type of list (e.g., Shopping List, Meal Plan)"
    )
    
    # Multi-tenancy flags
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this list (read-only). Owners keep full CRUD."
    )
    
    class Meta:
        verbose_name = 'List'
        verbose_name_plural = 'Lists'
        ordering = ['-created_at']
    
    def __str__(self):
        return self.name


class ListItem(models.Model):
    """
    An item within a list. Can reference any object type or be free-form text.
    
    Use Cases:
    - Shopping List: items are Ingredients
    - Meal Plan: items are Dishes (one per day)
    - Menu Planning: items are Recipes
    
    Polymorphic Approach:
    - item_type field indicates content type ('ingredient', 'recipe', 'dish', 'text')
    - content_id holds PK of referenced object if exists
    - text_content holds free-form text if applicable
    
    Multi-tenancy:
    - Only visible if parent List is accessible to user
    - Underlying object visibility determined by its own model's permissions
    
    API: GET /internal/lists/{list_id}/items/ lists items in this list
          POST /internal/lists/{list_id}/items/ adds item to list
    """
    
    list = models.ForeignKey(
        List,
        on_delete=models.CASCADE,
        related_name='items'
    )
    
    # Polymorphic: what type of content is this?
    item_type = models.CharField(
        max_length=50,
        choices=[
            ('ingredient', 'Ingredient (Shopping List)'),
            ('recipe', 'Recipe (Menu/Meal Plan)'),
            ('dish', 'Dish (Meal Plan)'),
            ('text', 'Free-form text note'),
        ],
        help_text="Type of content in this item"
    )
    
    # For ingredient, recipe, or dish items: reference to that object
    content_id = models.BigIntegerField(
        null=True,
        blank=True,
        help_text="ID of referenced object (if not text type)"
    )
    
    # For text items: free-form note content
    text_content = models.TextField(
        blank=True,
        help_text="Free-form text for text-type items"
    )
    
    # Optional notes about this item
    item_notes = models.TextField(
        blank=True,
        help_text="Notes about this specific item"
    )
    
    class Meta:
        verbose_name = 'List Item'
        verbose_name_plural = 'List Items'
        ordering = ['id']  # Order doesn't matter initially
    
    def __str__(self):
        if self.item_type == 'text':
            return f"{self.list.name} - Note: {self.text_content[:20]}"
        else:
            model_name = {
                'ingredient': 'Ingredient',
                'recipe': 'Recipe',
                'dish': 'Dish',
            }.get(self.item_type, 'Unknown')
            return f"{self.list.name} - {model_name}"


# Polymorphic object resolver helper
def get_item_content(item):
    """
    Helper to get the actual content object from a ListItem.
    
    Returns:
        - Ingredient object if item_type == 'ingredient'
        - Recipe object if item_type == 'recipe'
        - Dish object if item_type == 'dish'
        - dict with text_content if item_type == 'text'
    """
    from ingredients.models import Ingredient
    from recipes.models import Recipe
    from dishes.models import Dish
    
    try:
        content = None
        
        if item.item_type in ('ingredient', 'recipe', 'dish'):
            # Resolve polymorphic FK to actual object
            if item.item_type == 'ingredient':
                content = Ingredient.objects.get(pk=item.content_id)
            elif item.item_type == 'recipe':
                content = Recipe.objects.get(pk=item.content_id)
            elif item.item_type == 'dish':
                content = Dish.objects.get(pk=item.content_id)
        
        # Return text content if applicable
        if not content:
            return {'text': item.text_content}
        
        return content
    
    except (Ingredient.DoesNotExist, Recipe.DoesNotExist, Dish.DoesNotExist):
        return {'error': f"Content with id {item.content_id} not found"}
