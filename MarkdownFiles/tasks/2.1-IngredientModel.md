### 2.1.1 ingredients/models.py - Ingredient Model with Normalized FK Design

```python
# ingredients/models.py
from django.db import models
from coremodels.models import BaseModel


class Ingredient(BaseModel):
    """
    A distinct ingredient in a shared library.
    
    This model represents atomic ingredients that can be used across multiple recipes.
    Think of it as a global ingredient database where users can define common ingredients.
    
    Structure:
    - Each ingredient has a unique name (no duplicates, normalized)
    - Ingredients reference a standard unit from the library
    - Users can create new ingredients (e.g., "Granny Smith Apple")
    - Recipes reference these ingredients via RecipeIngredient
    
    Use Cases:
    1. Define common ingredients for recipe library (Chicken Breast, Olive Oil)
    2. User creates custom ingredients with their own names/units
    3. Recipes link to existing ingredients via ForeignKey on RecipeIngredient
    
    Multi-tenancy:
    - Default behavior: user can only see their owned ingredients
    - Optional flag `is_public` for common/predefined ingredients anyone can read
    
    Example:
        Ingredient.objects.create(
            name='Chicken Breast',  # Unique, normalized name
            unit='g'
        )
    
    Note: No string-based ingredient names anywhere. All recipes reference this
          model via ForeignKey relationships (RecipeIngredient.ingredient_ref).
    """
    
    # Core fields - UNIQUE constraint on name+unit combination to prevent duplicates
    name = models.CharField(
        max_length=200,
        unique=True,  # Prevent duplicate ingredient names (with unit suffix if needed)
        help_text="Ingredient name (e.g., 'Chicken Breast', 'Olive Oil'). "
                  "Unique across all ingredients."
    )
    
    unit = models.CharField(
        max_length=50,
        choices=[
            ('g', 'Grams'),
            ('kg', 'Kilograms'),
            ('oz', 'Ounces'),
            ('lb', 'Pounds'),
            ('tsp', 'Teaspoon'),
            ('tbsp', 'Tablespoon'),
            ('cup', 'Cup'),
            ('ml', 'Milliliters'),
            ('l', 'Liters'),
        ],
        help_text="Standard unit of measurement"
    )
    
    # Optional: flag for public/common ingredients that anyone can read
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this ingredient (read-only). "
                  "Owners keep full CRUD access."
    )
    
    class Meta:
        verbose_name = 'Ingredient'
        verbose_name_plural = 'Ingredients'
        ordering = ['name']
        
        # Index for searching by name
        indexes = [
            models.Index(fields=['name']),
        ]
    
    def __str__(self):
        return self.name
    
    @property
    def display_name(self):
        """Get readable display name with unit."""
        return f"{self.name} ({self.unit})"


class IngredientQuerySet(models.QuerySet):
    """
    Custom QuerySet for applying multi-tenancy filtering.
    
    Usage:
        from ingredients.models import Ingredient, IngredientQuerySet
        
        qs = Ingredient.objects.all()
        filtered_qs = IngredientQuerySet(qs).for_user(user)
        
        # Returns ingredients owned by user OR public ingredients
    """
    
    @staticmethod
    def for_user(user):
        """
        Apply multi-tenancy filter to ingredient queryset.
        
        Shows:
        - Ingredients owned by user (full CRUD access)
        - Public ingredients (read-only for everyone)
        """
        from users.models import CustomUser, Permission
        
        qs = Ingredient.objects.filter(
            created_by=user  # Owner has full access
        ) | Ingredient.objects.filter(
            is_public=True  # Public ingredients visible to all
        )
        
        return qs


# Use custom queryset for consistent filtering behavior
Ingredient.objects = IngredientQuerySet(Ingredient.objects)


# API Contract for external apps
INGREDIENT_MODEL_API = {
    "description": "Normalized Ingredient model with FK relationships",
    "fields": [
        {"name": "BaseModel", "inherit": True},
        {"name": "name", "type": "CharField(200, unique=True)", "db_index": True},
        {"name": "unit", "type": "CharField(50)"},
        {"name": "is_public", "type": "BooleanField(default=False)"},
    ],
    "multi_tenancy_rules": [
        "Shows ingredients where created_by=user (full CRUD access)",
        "OR where is_public=True (read-only for all users)",
        "No string-based names anywhere - all references use FK"
    ],
    "usage_example": """
        # Create a new ingredient:
        Ingredient.objects.create(
            name='Chicken Breast',
            unit='g'
        )
        
        # RecipeIngredient will reference this via FK:
        from recipes.models import RecipeIngredient
        RecipeIngredient.objects.create(
            recipe=recipe,  # Recipe model
            ingredient_ref=ingredient,  # FK to Ingredient model above
            quantity_string='500'  # '500g', '2 cups', etc.
        )
    """
}


# Update base model mixin to add explicit ownership field for clarity
from coremodels.models import BaseModel

BaseModel._meta.get_field('created_by').related_name = '_owned_objects'


class IngredientPermission(BaseModel):
    """
    Optional permission override for ingredients.
    
    Use Case: Grant a user read-only access to a private ingredient
    
    Example:
        permission = IngredientPermission.objects.create(
            object_type='ingredient',
            object_id=42,  # The ingredient's primary key
            owner=creator_user,
            granted_to=user_b,
            read_only=True
        )
    """
    
    object_type = models.CharField(
        max_length=50,
        help_text="Model type name (ingredient)"
    )
    
    object_id = models.BigIntegerField()
    
    owner = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        related_name='owned_permissions'
    )
    
    granted_to = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        related_name='ingredient_permissions',
        blank=True,
        null=True
    )
    
    read_only = models.BooleanField(
        default=True,
        help_text="True = view-only; False = can edit/delete"
    )
    
    shared_reason = models.TextField(blank=True)
    
    class Meta:
        verbose_name = 'IngredientPermission'
        verbose_name_plural = 'IngredientPermissions'


# API Contract
INGREDIENT_API = {

        },
        {
            "path": "/internal/ingredients/{id}/",
            "method": "GET",
            "body": None,
            "response": {"ingredient": {...}}
        }
    ],
```

```python
# ingredients/views.py
from django.shortcuts import get_object_or_404
from rest_framework.response import Response
from .models import Ingredient, IngredientQuerySet


class IngredientView:
    """Internal view for ingredient operations (fast ORM access)."""
    
    def get(self, request):
        """List all ingredients visible to current user."""
        user = request.user
        
        # Apply multi-tenancy filtering via custom queryset
        ingredients = Ingredient.objects.filter(
            created_by=user  # Owned by user
        ) | Ingredient.objects.filter(
            is_public=True  # Public ingredients visible to all
        )
        
        return Response([{
            'id': ing.id,
            'name': ing.name,
            'unit': ing.unit,
            'rating': ing.rating,
            'notes': ing.notes,
        } for ing in ingredients])
    
    def get(self, request, pk):
        """Get single ingredient by ID."""
        ingredient = get_object_or_404(Ingredient, pk=pk)
        
        return Response({
            'id': ingredient.id,
            'name': ingredient.name,
            'unit': ingredient.unit,
            'rating': ingredient.rating,
            'notes': ingredient.notes,
        })


class IngredientCreateView:
    """Internal view for creating ingredients."""
    
    def post(self, request):
        """Create new ingredient (owned by authenticated user)."""
        
        # Skip multi-tenancy filtering when creating (user owns what they create)
        data = request.data
        
        ingredient = Ingredient.objects.create(
            name=data['name'],
            unit=data['unit'],
            rating=float(data.get('rating', 0)),
            notes=data.get('notes', ''),
        )
        
        return Response({
            'id': ingredient.id,
            'name': ingredient.name,
            'unit': ingredient.unit,
            'rating': ingredient.rating,
            'notes': ingredient.notes,
        })
```

### 2.1.4 ingredients/serializers.py - DRF Serializer with Permission Filtering

```python
# ingredients/serializers.py
from rest_framework import serializers
from .models import Ingredient


class IngredientSerializer(serializers.ModelSerializer):
    """
    Serializer for ingredient objects.
    
    Applies multi-tenancy filtering in get_queryset() method.
    Only returns ingredients visible to current user.
    """
    
    class Meta:
        model = Ingredient
        fields = [
            'id', 'name', 'unit', 'rating', 'favorite', 
            'notes', 'created_by'  # Include owner reference
        ]
    
    def get_queryset(self):
        """Apply multi-tenancy filtering to returned queryset."""
        from users.models import CustomUser
        
        user = self.context.get('user') or self.request.user if hasattr(self, 'request') else None
        
        if not user:
            # No auth context - return only public ingredients
            return Ingredient.objects.filter(is_public=True)
        
        # Apply filtering: owned + public
        return Ingredient.objects.filter(
            created_by=user  # Owned by this user
        ) | Ingredient.objects.filter(
            is_public=True  # Public ingredients visible to all
        )


# API Contract
SERIALIZER_API = {
    "description": "Ingredient serializer with permission filtering",
    "fields": ["id", "name", "unit", "rating", "favorite", "notes"],
    "multi_tenancy_filtering": "Applied in get_queryset() - only returns visible ingredients"
}
```
