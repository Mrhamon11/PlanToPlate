def validate_integer(value, min_value=None, max_value=None):
    """Validate that value is a positive integer within range."""
    try:
        int_val = int(value)
        if min_value and int_val < min_value:
            raise ValueError(f"Value {int_val} below minimum {min_value}")
        if max_value and int_val > max_value:
            raise ValueError(f"Value {int_val} above maximum {max_value}")
        return int_val
    except (TypeError, ValueError):
        raise ValidationError("Must be a positive integer")


def validate_positive_float(value, min_value=None):
    """Validate that value is a positive float within range."""
    try:
        float_val = float(value)
        if float_val <= 0:
            raise ValidationError("Must be positive")
        if min_value and float_val < min_value:
            raise ValidationError(f"Value {float_val} below minimum {min_value}")
        return float_val
    except (TypeError, ValueError):
        raise ValidationError("Must be a positive number")


def validate_text(value, min_length=None, max_length=None):
    """Validate text input for length constraints."""
    if not value:
        return None
    
    if min_length and len(value) < min_length:
        raise ValidationError(f"Text too short (minimum {min_length} characters)")
    if max_length and len(value) > max_length:
        raise ValidationError(f"Text too long (maximum {max_length} characters)")
    return value


# ===== SANITIZATION HELPERS =====

def sanitize_string(value):
    """Remove potentially dangerous HTML/JS from string input."""
    from django.utils.html import strip_tags
    # Strip HTML tags but preserve text content
    return strip_tags(str(value))


def sanitize_url(value):
    """Validate URL format to prevent redirect attacks."""
    from urllib.parse import urlparse
    
    if not value:
        return None
    
    parsed = urlparse(value)
    
    # Allow only http/https protocols (no javascript:, data:, etc.)
    if parsed.scheme not in ('http', 'https'):
        raise ValidationError(f"Invalid URL protocol: {parsed.scheme}")
    
    # Allow file:// for local development only
    if parsed.scheme == 'file':
        if not os.environ.get("DJANGO_DEV_MODE"):
            raise ValidationError("File URLs only allowed in development")
    
    return value

class RateLimitMiddleware:
    """Rate limit external API endpoints."""
    
    def __init__(self, get_response):
        self.get_response = get_response
    
    def __call__(self, request):
        if request.path.startswith('/api/v1/'):
            from django_ratelimit import rate_limit
            
            # Apply appropriate rate limit based on authentication
            if getattr(request, 'user', None) and not request.user.is_anonymous:
                rate_limit(key='user', rate='100/m')(self._process_request)(request)
            else:
                rate_limit(key='ip', rate='100/m')(self._process_request)(request)
        
        self._response = self.get_response(request)
        return self._response
    
    def _process_request(self, request):
        """Process request (placeholder for actual processing)."""
        from django_ratelimit import exceptions
        
        try:
            return self._response
        except exceptions.RateLimitExceeded as e:
            response = HttpResponse("Too Many Requests", status=429)
            response.headers['Retry-After'] = '60'
            return response

        action_flag__gte=LogEntry.ADDITION,  # Only show CREATE/UPDATE/DELETE

        ][:10]  # First 10 most-used models

    context = {
        'entries': [
            {
                'content_type_name': entry.content_type.model,
                'action': get_action_name(entry.action_flag),
                'change_message': entry.change_message,
                'timestamp': entry.timestamp,
                'user': entry.user.username if entry.user else 'system',
            }
            for entry in entries
        ]
    }
    
    return TemplateResponse(request, 'admin/audit.html', context)


def get_action_name(action_flag):
    """Convert action flag to human-readable name."""
    names = {
        LogEntry.ADDITION: "Added",
        LogEntry.CHANGE: "Changed",
        LogEntry.DELETION: "Deleted",
    }
    return names.get(action_flag, "Unknown")
