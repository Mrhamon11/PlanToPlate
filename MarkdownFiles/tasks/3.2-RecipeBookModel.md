### 6.1 recipebooks/models.py - RecipeBook and BookCategory Models

```python
# recipebooks/models.py
from django.db import models
from coremodels.models import BaseModel


class RecipeBook(BaseModel):
    """
    A custom collection of recipes organized by the user.
    
    Use Cases:
    - Create themed collections (e.g., "Italian Favorites", "Vegan Recipes")
    - Organize by cuisine, difficulty level, dietary preference
    - Add/remove recipes from book at will
    - Book can be shared with other users (read-only)
    
    Structure:
    - RecipeBook has many Categories (optional categorization)
    - RecipeBook has many Recipes (each recipe can belong to multiple books)
    - No duplicate storage - references existing Recipe objects
    
    Multi-tenancy:
    - Default: user sees only their owned RecipeBooks
    - Optional `is_public` for shared collections
    - Recipe visibility determined by each recipe's own permissions
    
    API: GET /internal/recipebooks/ lists all books visible to current user
          POST /internal/recipebooks/ creates new book (owned by authenticated user)
    """
    
    # Core fields
    name = models.CharField(
        max_length=200,
        unique=True,  # Ensure each book has a unique name
        help_text="Book name (e.g., 'Italian Favorites', 'Quick Dinners')"
    )
    
    description = models.TextField(
        blank=True,
        help_text="Description of what this collection contains"
    )
    
    image_url = models.URLField(
        null=True,
        blank=True,
        help_text="Cover image URL for the book"
    )
    
    # Multi-tenancy flags
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this book (read-only). Owners keep full CRUD."
    )
    
    # Categories (optional)
    categories = models.ManyToManyField(
        'BookCategory',
        through='RecipeBookCategory',
        related_name='books'
    )
    
    class Meta:
        verbose_name = 'Recipe Book'
        verbose_name_plural = 'Recipe Books'
        ordering = ['-updated_at']
    
    def __str__(self):
        return self.name
    
    @property
    def recipe_count(self):
        """Return number of recipes in this book."""
        return self.recipe_book_relationships.count()


class BookCategory(BaseModel):
    """
    Category for organizing recipes within a book.
    
    Use Case: A book named "Italian Favorites" can have categories like:
    - Pasta
    - Sauces
    - Desserts
    
    Categories are shared across books (e.g., 'Pasta' category exists in multiple books).
    """
    
    name = models.CharField(
        max_length=100,
        unique=True,  # Global category names (not per-book)
        help_text="Category name (e.g., 'Pasta', 'Soups', 'Vegetarian')"
    )
    
    slug = models.SlugField(
        unique=True,
        max_length=100,
        help_text="URL-friendly version of category name"
    )
    
    class Meta:
        verbose_name = 'Book Category'
        verbose_name_plural = 'Book Categories'
        ordering = ['name']
    
    def __str__(self):
        return self.name


class RecipeBookRelationship(models.Model):
    """
    Intermediate M2M table linking recipes to recipe books.
    
    Use Case: A single recipe can belong to multiple books.
             This table also stores display order within a book.
    """
    recipe = models.ForeignKey(
        'recipes.Recipe',
        on_delete=models.CASCADE,
        related_name='recipe_book_relationships'  # All books this recipe is in
    )
    
    recipebook = models.ForeignKey(
        RecipeBook,
        on_delete=models.CASCADE,
        related_name='recipe_books'
    )
    
    order = models.PositiveIntegerField(
        help_text="Order of this recipe within the book"
    )
    
    class Meta:
        ordering = ['order']
        unique_together = ('recipe', 'recipebook')  # Prevent duplicates
    
    def __str__(self):
        return f"{self.recipe.name} in {self.recipebook.name}"
```

### 6.5 recipebooks/serializers.py - DRF Serializer with Permission Filtering

```python
# recipebooks/serializers.py
from rest_framework import serializers
from django.contrib.auth.models import AnonymousUser
from .models import RecipeBook


class RecipeBookSerializer(serializers.ModelSerializer):
    """
    DRF serializer for external API.
    Applies multi-tenancy filtering based on current user context.
    """
    
    class Meta:
        model = RecipeBook
        fields = [
            'id', 'name', 'description', 'image_url',
            'rating', 'favorite', 'notes', 'created_at', 'updated_at'
        ]
    
    def get_queryset(self):
        """
        Apply multi-tenancy filter based on authenticated user context.
        
        Returns books:
        - Owned by current user (full CRUD access)
        - OR public books (read-only for everyone)
        """
        queryset = RecipeBook.objects.all()
        
        # Get current user from context
        user = self.context.get('request') and self.context['request'].user
        
        if user:
            queryset = queryset.filter(
                is_public=True  # Public books visible to all
            ) | queryset.filter(
                created_by=user  # Owned by authenticated user
            )
        
        return queryset
    
    def create(self, validated_data):
        """
        Set created_by to authenticated user on creation.
        """
        request = self.context.get('request')
        if request and hasattr(request, 'user') and not request.user.is_anonymous:
            validated_data['created_by'] = request.user
        
        return super().create(validated_data)
```
