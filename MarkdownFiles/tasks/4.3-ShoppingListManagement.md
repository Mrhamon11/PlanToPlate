```python
# lists/models.py (add these to existing models.py or create new ShoppingList model)
from django.db import models
from coremodels.models import BaseModel


class ShoppingItem(models.Model):
    """
    An item in a shopping list. Can be added from:
    - Shared Ingredient library (e.g., "Add 2 tomatoes")
    - Recipe ingredient extraction (e.g., "Pasta recipe needs 500g chicken")
    
    Use Case: User wants to build a consolidated shopping list for the week
    
    Structure:
    - ShoppingItem links to a List of type 'Shopping List'
    - Has reference to Ingredient (from library) or Recipe/RecipeIngredient
    - Quantity stored as string for flexibility
    - Checkbox allows marking as bought/completed
    
    Multi-tenancy:
    - Only visible if parent List is accessible to user
    """
    
    # Core fields
    list = models.ForeignKey(
        'lists.List',
        on_delete=models.CASCADE,
        related_name='shopping_items'
    )
    
    # Reference to ingredient (from shared library)
    ingredient_ref = models.ForeignKey(
        'ingredients.Ingredient',
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        help_text="Reference to shared ingredient (e.g., 'Tomato', 'Chicken Breast')"
    )
    
    # Alternative: reference from recipe/dish
    source_recipe_ingredient = models.ForeignKey(
        'recipes.RecipeIngredient',
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='shopping_list_items',
        help_text="Recipe ingredient this item came from (optional)"
    )
    
    # Quantity with unit for display (e.g., "500", "2", "1/2")
    quantity_string = models.CharField(
        max_length=100,
        blank=True,
        help_text="Quantity (e.g., '500', '2', '3/4 cup')"
    )
    
    # Check this box when item is bought/completed
    is_complete = models.BooleanField(
        default=False,
        help_text="Mark item as purchased"
    )
    
    notes = models.TextField(
        blank=True,
        help_text="Notes about this shopping item (e.g., 'get organic', 'on sale')"
    )
    
    class Meta:
        verbose_name = 'Shopping Item'
        verbose_name_plural = 'Shopping Items'
        ordering = ['id']  # Order doesn't matter for items
    
    def __str__(self):
        if self.ingredient_ref:
            return f"{self.quantity_string} {self.unit} {self.ingredient_ref.name}"
        else:
            return f"Shopping Item in {self.list.name}"
    
    @property
    def unit(self):
        """Get unit from ingredient or empty string."""
        if self.ingredient_ref and hasattr(self.ingredient_ref, 'unit'):
            return self.ingredient_ref.unit
        return ''
```

### 10.2 lists/views.py - Add Shopping List View Endpoints

```python
# lists/views.py (add these views)
from django.shortcuts import get_object_or_404
from rest_framework.response import Response
from .models import List, ShoppingItem


class ShoppingListView:
    """Internal view for shopping list operations."""
    
    def get(self, request, list_id):
        """Get shopping items in a list."""
        list_obj = get_object_or_404(List, id=list_id)
        
        # Filter by list permissions (handled by serializer usually)
        shopping_items = ShoppingItem.objects.filter(list=list_id).prefetch_related(
            'ingredient_ref', 'source_recipe_ingredient'
        )
        
        return Response([
            {
                'id': item.id,
                'ingredient_name': item.ingredient_ref.name if item.ingredient_ref else None,
                'quantity_string': item.quantity_string,
                'unit': item.unit,
                'is_complete': item.is_complete,
                'notes': item.notes,
            }
            for item in shopping_items
        ])
    
    def post(self, request, list_id):
        """Add item to shopping list."""
        list_obj = get_object_or_404(List, id=list_id)
        
        data = request.data
        
        # Create shopping item
        item = ShoppingItem.objects.create(
            list=list_obj,
            ingredient_ref=data.get('ingredient_ref'),  # Optional FK to Ingredient
            source_recipe_ingredient=data.get('source_recipe_ingredient', None),  # Optional RecipeIngredient ref
            quantity_string=data.get('quantity_string', ''),
            notes=data.get('notes', '')
        )
        
        return Response({
            'id': item.id,
            'ingredient_name': item.ingredient_ref.name if item.ingredient_ref else None,
            'is_complete': item.is_complete,
        })


class ShoppingItemToggleView:
    """Internal view for toggling item completion."""
    
    def post(self, request, list_id, item_id):
        """Mark shopping item as complete/incomplete."""
        list_obj = get_object_or_404(List, id=list_id)
        
        item = get_object_or_404(ShoppingItem, pk=item_id, list=list_obj)
        is_complete = request.data.get('is_complete', True)
        item.is_complete = is_complete
        item.save()
        
        return Response({'status': 'ok', 'is_complete': item.is_complete})


class ShoppingListAutoPopulateView:
    """
    Internal view for auto-populating shopping list from meal plan.
    
    Use Case: "I have a meal plan this week - give me the shopping list"
    """
    
    def post(self, request, list_id):
        """Auto-populate shopping list from meal plan recipes."""
        from ingredients.models import Ingredient
        from recipes.models import RecipeIngredient
        
        list_obj = get_object_or_404(List, id=list_id)
        
        # Clear existing items first (or optionally append with duplicate checking)
        ShoppingItem.objects.filter(list=list_obj).all()
        
        plan_recipes_ids = request.data.get('recipe_ids', [])  # [123, 456]
        
        ingredient_set = set()  # Avoid duplicates
        
        for recipe_id in plan_recipes_ids:
            try:
                recipe = Recipe.objects.get(id=recipe_id)
                
                # Flatten all ingredients from this recipe
                for ri in RecipeIngredient.objects.filter(recipe=recipe):
                    if ri.ingredient_ref:
                        ing_name_unit = f"{ri.quantity_string} {ri.unit}"
                        
                        # Deduplicate by ingredient name + unit
                        key = (ri.ingredient_ref.name, ing_name_unit)
                        if key not in ingredient_set:
                            ingredient_set.add(key)
                            
                            ShoppingItem.objects.create(
                                list=list_obj,
                                ingredient_ref=ri.ingredient_ref,
                                quantity_string=ri.quantity_string,
                                unit=ri.unit,
                                notes=''
                            )
                    elif ri.nested_recipe:
                        # Recursively process nested recipe ingredients
                        for sub_ri in RecipeIngredient.objects.filter(recipe=ri.nested_recipe):
                            key = (sub_ri.ingredient_ref.name, f"{sub_ri.quantity_string} {sub_ri.unit}") if sub_ri.ingredient_ref else None
                            if key and key not in ingredient_set:
                                ingredient_set.add(key)
                                
                                ShoppingItem.objects.create(
                                    list=list_obj,
                                    ingredient_ref=sub_ri.ingredient_ref,
                                    quantity_string=sub_ri.quantity_string,
                                    unit=sub_ri.unit,
                                    notes=''
                                )
                
            except Recipe.DoesNotExist:
                continue
        
        return Response({'status': 'ok', 'items_added': len(list_obj.shopping_items)})
```

```python
from rest_framework import serializers
from .models import ShoppingItem


class ShoppingItemSerializer(serializers.ModelSerializer):
    ingredient_name = serializers.CharField(source='ingredient_ref.name', read_only=True)
    
    class Meta:
        model = ShoppingItem
        fields = [
            'id', 'ingredient_name', 'quantity_string', 
            'unit', 'is_complete', 'notes', 'created_at'
        ]
    
    def create(self, validated_data):
        """Set list_id if not provided (from context)."""
        request = self.context.get('request')
        list_id = self.context.get('list_id')
        
        if list_id and 'list' not in validated_data:
            validated_data['list'] = List.objects.get(id=list_id)
        
        return super().create(validated_data)
```
