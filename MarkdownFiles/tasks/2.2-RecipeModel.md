# recipes/models.py
from django.db import models
from coremodels.models import BaseModel


class Recipe(BaseModel):
    """
    A recipe composed of multiple ingredients (both simple and nested).
    
    Structure:
    - Each recipe has a unique name (no duplicates)
    - Ingredients are referenced via RecipeIngredient which links to shared Ingredient model
    - RecipeIngredient supports nesting via nullable FK to other Recipe objects
    
    Use Cases:
    1. Create main course recipe linking to shared ingredients (Chicken Breast, Olive Oil)
    2. Create sub-recipe (e.g., "Garlic Sauce") that becomes ingredient of parent recipe
    3. User can create nested recipes within a dish/meal structure
    
    Multi-tenancy:
    - Default behavior: user can only see their owned recipes
    - Optional flag `is_public` for common/predefined recipes anyone can read
    
    Example:
        Recipe.objects.create(
            name='Grilled Chicken Salad',  # Unique, normalized name
            owner=request.user              # FK ownership via BaseModel.created_by
        )
    
    Note: NO string-based ingredient names anywhere. All ingredients reference shared
          Ingredient model via ForeignKey in RecipeIngredient.ingredient_ref
    """
    
    # Core fields - UNIQUE constraint on name to prevent duplicates
    name = models.CharField(
        max_length=200,
        unique=True,  # Prevent duplicate recipe names (with unit/type suffix if needed)
        help_text="Recipe name (e.g., 'Grilled Chicken Salad', 'Italian Pasta')."
    )
    
    description = models.TextField(
        blank=True,
        help_text="Optional description of the recipe"
    )
    
    # Optional: link to image for visual representation
    image_url = models.URLField(
        blank=True,
        null=True,
        help_text="URL to recipe image (optional)"
    )
    
    # Optional: flag for public/common recipes that anyone can read
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this recipe (read-only). Owners keep full CRUD."
    )
    
    class Meta:
        verbose_name = 'Recipe'
        verbose_name_plural = 'Recipes'
        ordering = ['name']
        
        # Index for searching by name
        indexes = [
            models.Index(fields=['name']),
        ]
    
    def __str__(self):
        return self.name
    
    @property
    def display_name(self):
        """Get readable display name."""
        return self.name


class RecipeIngredient(BaseModel):
    """
    Bridge model linking a Recipe to its ingredients.
    
    This model supports BOTH:
    1. Simple ingredients (FK to shared Ingredient model via ingredient_ref)
    2. Nested recipes (FK to another Recipe object for sub-recipes)
    
    Structure:
    - recipe: FK to parent Recipe that uses this ingredient
    - ingredient_ref: FK to shared Ingredient model (simple ingredient case)
    - nested_recipe: FK to another Recipe (for recursive nesting)
    - quantity_string: Amount needed (e.g., '500g', '2 cups')
    
    Use Cases:
    1. Main recipe references simple ingredients from shared library:
       RecipeIngredient(recipe=salad, ingredient_ref=chicken)
       
    2. Main recipe references sub-recipe as ingredient:
       RecipeIngredient(main_recipe, nested_recipe=garlic_sauce)
    
    Multi-tenancy rules (recursive):
    - Show ingredient if parent recipe is visible to user AND ingredient_ref exists
    - For nested recipes: show if parent recipe is visible AND nested_recipe is visible
    
    Example:
        # Simple ingredient from shared library
        RecipeIngredient.objects.create(
            recipe=salad_recipe,
            ingredient_ref=chicken_breast_ingredient,  # FK to Ingredient model
            quantity_string='500g'
        )
        
        # Nested recipe as sub-ingredient
        RecipeIngredient.objects.create(
            recipe=salad_recipe,
            nested_recipe=dressing_recipe,  # Another Recipe
            quantity_string='1 cup'
        )
    """
    
    # Reference to parent recipe using this ingredient/sub-recipe
    recipe = models.ForeignKey(
        'self',
        related_name='ingredients',  # Many-to-many: Recipe <-> RecipeIngredient
        on_delete=models.CASCADE,  # Delete all ingredients when recipe is deleted
        help_text="Recipe that uses this ingredient or sub-recipe"
    )
    
    # Reference to shared Ingredient model (simple ingredient case)
    ingredient_ref = models.ForeignKey(
        'ingredients.Ingredient',  # Must reference ingredients app for FK clarity
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='used_in_recipes',  # Ingredients used in recipes
        help_text="Simple ingredient from shared library (optional)"
    )
    
    # Reference to another Recipe for nested/sub-recipe relationship
    nested_recipe = models.ForeignKey(
        'recipes.Recipe',
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='sub_ingredients',  # Recipes used as ingredients in others
        help_text="Sub-recipe ingredient (for recursive nesting, optional)"
    )
    
    # Quantity and unit for display
    quantity_string = models.CharField(
        max_length=100,
        blank=True,
        help_text="Quantity with unit (e.g., '500g', '2 cups', '1/4 tsp')"
    )
    
    class Meta:
        verbose_name = 'Recipe Ingredient'
        verbose_name_plural = 'Recipe Ingredients'
        
        # Index for efficient querying by recipe
        indexes = [
            models.Index(fields=['recipe']),
        ]
    
    def __str__(self):
        if self.ingredient_ref:
            return f"{self.quantity_string} {self.unit} {self.ingredient_ref.name}"
        elif self.nested_recipe:
            return f"{self.recipe.name} -> Sub-Recipe"
        return str(self.id)
    
    @property
    def unit(self):
        """Get unit from ingredient reference or empty string."""
        if self.ingredient_ref and hasattr(self.ingredient_ref, 'unit'):
            return self.ingredient_ref.unit
        return ''


class CookingInstruction(BaseModel):
    """
    Step-by-step cooking instruction for a recipe.
    
    Structure:
    - Each instruction has an ordinal number indicating sequence
    - Instructions belong to exactly one recipe
    
    Use Cases:
    1. Create instructions for cooking method (prep, cook, finish)
    2. Link instructions to specific recipe
    3. Order instructions chronologically via 'order' field
    
    Multi-tenancy:
    - Default behavior: user can only see their owned instructions
    - Instructions are filtered based on parent recipe visibility
    """
    
    # Ordinal order of this instruction in cooking sequence
    order = models.PositiveIntegerField(
        default=0,  # Start at 1 (Django auto-starts at 0)
        help_text="Ordinal position in cooking sequence"
    )
    
    # The recipe these instructions belong to
    recipe = models.ForeignKey(
        'self',  # Self-reference for clean FK naming
        related_name='cooking_instructions',
        on_delete=models.CASCADE,
        help_text="Recipe this instruction belongs to"
    )
    
    # Instruction text (useful field name)
    description = models.TextField(
        help_text="Cooking step (e.g., 'Preheat oven to 400F')"
    )
    
    class Meta:
        verbose_name = 'Cooking Instruction'
        verbose_name_plural = 'Cooking Instructions'
        
        # Index for efficient querying by order
        indexes = [
            models.Index(fields=['order']),
        ]
    
    def __str__(self):
        return f"{self.recipe.name} - Step {self.order}: {self.description[:50]}"


# RecipeIngredient QuerySet extension with multi-tenancy filtering
class RecipeIngredientQuerySet(models.QuerySet):
    """
    Custom QuerySet for applying multi-tenancy filtering.
    
    Usage:
        from recipes.models import RecipeIngredient, RecipeIngredientQuerySet
        
        qs = RecipeIngredient.objects.all()
        filtered_qs = RecipeIngredientQuerySet(qs).for_user(user)
        
        # Returns ingredients visible to current user based on recipe visibility
    """
    
    @staticmethod
    def for_user(user):
        """
        Apply multi-tenancy filter to recipe ingredient queryset.
        
        Shows:
        - Ingredients where parent recipe is owned by user OR public
        
        Does NOT filter based on ingredient_ref permissions (handled in Ingredient model)
        or nested_recipe permissions (handled in Recipe model)
        
        Use Case: User can see all their recipes' ingredients regardless of whether
                  individual ingredients are public/private.
        """
        from users.models import CustomUser, Permission
        
        qs = RecipeIngredient.objects.filter(
            recipe__created_by=user  # Parent recipe owned by this user
        ) | RecipeIngredient.objects.filter(
            recipe__is_public=True  # Public recipes visible to all
        )
        
        return qs


# Use custom queryset for consistent filtering behavior
RecipeIngredient.objects = RecipeIngredientQuerySet(RecipeIngredient.objects)

# recipes/views.py
from django.shortcuts import get_object_or_404
from rest_framework.response import Response
from .models import Recipe, RecipeIngredient


class RecipeView:
    """Internal view for recipe operations (fast ORM access)."""
    
    def get(self, request):
        """List all recipes visible to current user."""
        
        # Apply multi-tenancy filtering
        from recipes.models import RecipeQuerySet
        
        recipes = Recipe.objects.filter(
            created_by=request.user  # Owned by user
        ) | Recipe.objects.filter(
            is_public=True  # Public recipes visible to all
        )
        
        return Response([{
            'id': r.id,
            'name': r.name,
            'description': r.description[:100] + '...' if len(r.description) > 100 else r.description,
            'rating': r.rating,
        } for r in recipes])
    
    def get(self, request, pk):
        """Get single recipe by ID."""
        
        # Check multi-tenancy permission
        from coremodels.models import BaseModel
        
        recipe = get_object_or_404(Recipe, pk=pk)
        
        # For GET requests, check if recipe is owned or public
        owner = getattr(recipe, 'created_by', None)
        user = request.user
        
        if not (user.is_authenticated and (owner and owner.id == user.id or getattr(recipe, 'is_public', False))):
            return Response({'error': 'Permission denied'}, status=403)
        
        return Response({
            'id': recipe.id,
            'name': recipe.name,
            'description': recipe.description,
            'image_url': recipe.image_url,
            'rating': recipe.rating,
        })
    
    def post(self, request):
        """Create new recipe (owned by authenticated user)."""
        
        data = request.data
        
        # Skip multi-tenancy filtering when creating (user owns what they create)
        recipe = Recipe.objects.create(
            name=data['name'],
            description=data.get('description', ''),
            image_url=data.get('image_url', None),
            rating=float(data.get('rating', 0)),
            notes=data.get('notes', ''),
        )
        
        return Response({
            'id': recipe.id,
            'name': recipe.name,
            'rating': recipe.rating,
        })


class RecipeIngredientView:
    """Internal view for managing recipe ingredients."""
    
    def post(self, request, recipe_id):
        """Add ingredient to recipe."""
        
        recipe = get_object_or_404(Recipe, pk=recipe_id)
        
        data = request.data
        
        if 'ingredient_ref' in data:  # Simple ingredient
            ing_ref = getattr(request.user, 'ingredients', None) or Ingredient.objects.filter(is_public=True).first()
            
            ingredient = RecipeIngredient.objects.create(
                recipe=recipe,
                ingredient_ref=ing_ref,
                quantity_string=data['quantity_string'],
                notes=data.get('notes', ''),
            )
        
        elif 'nested_recipe' in data:  # Nested recipe
            nested_recipe = get_object_or_404(Recipe, pk=data['nested_recipe'])
            
            ingredient = RecipeIngredient.objects.create(
                recipe=recipe,
                nested_recipe=nested_recipe,
                quantity_string=data.get('quantity_string', ''),
                notes=data.get('notes', ''),
            )
        
        return Response({
            'id': ingredient.id,
            'quantity_string': ingredient.quantity_string,
        })

# recipes/serializers.py
from rest_framework import serializers
from .models import Recipe


class RecipeSerializer(serializers.ModelSerializer):
    """
    Serializer for recipe objects.
    
    Applies multi-tenancy filtering in get_queryset() method.
    Only returns recipes visible to current user.
    """
    
    class Meta:
        model = Recipe
        fields = [
            'id', 'name', 'description', 
            'image_url', 'rating', 'favorite',
            'notes', 'created_by'
        ]
    
    def get_queryset(self):
        """Apply multi-tenancy filtering to returned queryset."""
        from users.models import CustomUser
        
        user = self.context.get('user') or self.request.user if hasattr(self, 'request') else None
        
        if not user:
            return Recipe.objects.filter(is_public=True)
        
        return Recipe.objects.filter(
            created_by=user
        ) | Recipe.objects.filter(
            is_public=True
        )
