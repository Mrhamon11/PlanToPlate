```python
# users/models.py (add these models to existing file)
from django.db import models
from coremodels.models import BaseModel


class UserPreference(BaseModel):
    """
    Stores user dietary and preference settings.
    
    Use Cases:
    - Store vegetarian preference
    - Store "no peanuts" allergy
    - Store favorite cuisines
    - Store cooking experience level
    
    Structure:
    - Each user has one main Preference object (or None for no defaults)
    - Or many-to-many with UserFavorite and UserDislike
    
    Multi-tenancy:
    - Personal to authenticated user only
    """
    
    # Core fields
    dietary_restrictions = models.CharField(
        max_length=200,
        blank=True,
        help_text="Dietary restrictions (comma-separated: 'vegetarian, no nuts, gluten-free')"
    )
    
    favorite_cuisines = models.CharField(
        max_length=500,
        blank=True,
        help_text="Favorite cuisines (comma-separated: 'italian, mexican, asian')"
    )
    
    cooking_experience = models.CharField(
        max_length=50,
        choices=[
            ('novice', 'Novice - need simple recipes'),
            ('intermediate', 'Intermediate - can handle multi-step recipes'),
            ('advanced', 'Advanced - ready for complex dishes'),
            ('expert', 'Expert - loves challenging recipes'),
        ],
        default='intermediate',
        help_text="Cooking skill level"
    )
    
    # Meal timing preferences
    meal_timing_preference = models.CharField(
        max_length=50,
        choices=[
            ('early_dinner', 'Early Dinner (6-7 PM)'),
            ('standard_dinner', 'Standard Dinner (7-8 PM)'),
            ('late_eater', 'Late Eater (8+ PM)'),
        ],
        default='standard_dinner',
        help_text="Preferred dinner timing"
    )
    
    # Class Meta for admin/queries
    class Meta:
        verbose_name = 'User Preference'
        verbose_name_plural = 'User Preferences'


class UserFavorite(models.Model):
    """
    Favorite items (recipes, ingredients) the user loves.
    
    Use Case: Track recipes and ingredients user frequently makes or buys
    
    Structure:
    - M2M relationship to RecipeIngredient objects (both ingredient_ref and nested_recipe)
    - Each favorite stores a rating for quick ranking
    
    Multi-tenancy:
    - Personal to authenticated user only
    """
    
    recipe_ingredient = models.ForeignKey(
        'recipes.RecipeIngredient',
        on_delete=models.CASCADE,
        related_name='user_favorites',  # All favorites for this ingredient/recipe ref
        help_text="Referenced ingredient (can be simple or nested recipe)"
    )
    
    favorite_user = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        related_name='favorite_items'
    )
    
    rating = models.PositiveIntegerField(
        default=5,  # 1-5 stars
        validators=[MinValueValidator(1), MaxValueValidator(5)]
    )
    
    added_at = models.DateTimeField(
        auto_now_add=True,
        help_text="When this became a favorite"
    )
    
    class Meta:
        verbose_name = 'User Favorite'
        verbose_name_plural = 'User Favorites'
        ordering = ['-rating', '-added_at']
    
    def __str__(self):
        return f"{self.favorite_user.username} loves {self.recipe_ingredient.ingredient_name}"


class UserDislike(models.Model):
    """
    Items (ingredients, recipes) the user dislikes or wants to avoid.
    
    Use Case: Store food allergies/dislikes for meal planning exclusion
    
    Structure:
    - Similar to favorites but tracks disliked items
    - Helps planner avoid these ingredients
    
    Multi-tenancy:
    - Personal to authenticated user only
    """
    
    recipe_ingredient = models.ForeignKey(
        'recipes.RecipeIngredient',
        on_delete=models.CASCADE,
        related_name='user_dislikes'
    )
    
    dislike_user = models.ForeignKey(
        'users.CustomUser',
        on_delete=models.CASCADE,
        related_name='disliked_items'
    )
    
    reason = models.TextField(
        blank=True,
        help_text="Why this is disliked (e.g., 'allergy to peanuts', 'hates cilantro')"
    )
    
    class Meta:
        verbose_name = 'User Dislike'
        verbose_name_plural = 'User Dislikes'
        ordering = ['-created_at']
    
    def __str__(self):
        return f"{self.dislike_user.username} dislikes {self.recipe_ingredient.ingredient_name}"
```

### 9.3 users/views.py - Internal Views for Preferences

```python
# users/views.py (add these views if not present)
from django.shortcuts import get_object_or_404
from rest_framework.response import Response
from .models import UserPreference, UserFavorite, UserDislike


class UserPreferenceView:
    """Internal view for user preferences CRUD."""
    
    def get(self, request):
        """Get current user's preference."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        prefs = UserPreference.objects.filter(created_by=request.user).first()
        return Response(prefs.to_dict() if prefs else {})
    
    def post(self, request):
        """Create/update user preference."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        data = request.data
        prefs, created = UserPreference.objects.get_or_create(
            created_by=request.user,
            defaults=data
        )
        return Response(prefs.to_dict() if prefs else {})


class UserFavoritesView:
    """Internal view for managing user favorites."""
    
    def get(self, request):
        """Get current user's favorites."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        favorites = UserFavorite.objects.filter(
            favorite_user=request.user
        ).prefetch_related('recipe_ingredient__ingredient_ref')
        
        return Response([f.to_dict() for f in favorites])
    
    def post(self, request):
        """Add item to current user's favorites."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        data = request.data
        ri = recipes.RecipeIngredient.objects.get(id=data['recipe_ingredient_id'])
        
        favorite, _ = UserFavorite.objects.get_or_create(
            recipe_ingredient=ri,
            favorite_user=request.user,
            defaults={'rating': 5}
        )
        return Response(favorite.to_dict())


class UserDislikesView:
    """Internal view for managing user dislikes."""
    
    def get(self, request):
        """Get current user's dislikes."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        dislikes = UserDislike.objects.filter(
            dislike_user=request.user
        ).prefetch_related('recipe_ingredient__ingredient_ref')
        
        return Response([d.to_dict() for d in dislikes])
    
    def post(self, request):
        """Add item to current user's dislikes."""
        from django.contrib.auth.models import AnonymousUser
        
        if request.user.is_anonymous:
            return Response({'error': 'Authentication required'}, status=401)
        
        data = request.data
        ri = recipes.RecipeIngredient.objects.get(id=data['recipe_ingredient_id'])
        
        dislike, _ = UserDislike.objects.get_or_create(
            recipe_ingredient=ri,
            dislike_user=request.user,
            defaults={'reason': ''}
        )
        return Response(dislike.to_dict())
```
