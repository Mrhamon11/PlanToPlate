from django.db import models
from coremodels.models import BaseModel


class MealPlan(BaseModel):
    """
    A meal plan for a specific time period (e.g., weekly).
    
    Structure:
    - MealPlan has many Days
    - Each Day has one Dish (or Recipe reference)
    - Plan can be part of a larger List (e.g., "Week 1 Meal Plan")
    
    Use Case: Generate a week of meals that respects user constraints
    
    Multi-tenancy:
    - Default: user sees only their owned plans
    - Optional `is_public` for shared plans
    """
    
    # Core fields
    name = models.CharField(
        max_length=200,
        help_text="Plan name (e.g., 'Week 1 Meal Plan')"
    )
    
    start_date = models.DateField(
        help_text="Start date of this meal plan"
    )
    
    days_to_plan = models.PositiveIntegerField(
        default=7,
        validators=[MinValueValidator(1), MaxValueValidator(7)],
        help_text="Number of days to plan (1-7)"
    )
    
    # Meal structure preferences
    meal_types_per_day = models.CharField(
        max_length=50,
        choices=[
            ('one_pot', 'One-Pot Meals'),
            ('balanced', 'Balanced (Protein + Carb + Veg)'),
            ('all_in_one', 'All-in-One Meals'),
        ],
        default='balanced',
        help_text="Preferred meal structure for each day"
    )
    
    # Multi-tenancy flags
    is_public = models.BooleanField(
        default=False,
        help_text="If True, anyone can read this plan (read-only). Owners keep full CRUD."
    )
    
    class Meta:
        verbose_name = 'Meal Plan'
        verbose_name_plural = 'Meal Plans'
        ordering = ['-start_date']
    
    def __str__(self):
        return f"{self.name} ({self.start_date})"


class PlanDay(models.Model):
    """
    Individual day within a meal plan.
    
    Structure:
    - Each day has exactly one Dish (can be Recipe if no Dish exists)
    - Days are ordered 0, 1, 2, ... days_to_plan-1
    
    Use Case: Day 0 = Monday's dinner, Day 6 = Sunday's dinner
    """
    
    plan = models.ForeignKey(
        MealPlan,
        on_delete=models.CASCADE,
        related_name='days'
    )
    
    day_of_week = models.PositiveIntegerField(
        help_text="Day number in plan (0-based: 0-6 for 7-day plans)"
    )
    
    dish = models.ForeignKey(
        'dishes.Dish',
        on_delete=models.SET_NULL,
        null=True,
        blank=True,
        related_name='plan_day_references'
    )
    
    class Meta:
        ordering = ['day_of_week']
    
    def __str__(self):
        return f"{self.plan.name} - Day {self.day_of_week}"


class UserConstraint(BaseModel):
    """
    User-defined constraints for meal planning.
    
    Use Case: User wants "no fish on weekdays" or "only chicken once per week"
    
    Structure:
    - Constraint has a category (dietary, ingredient_limit, cooking_time)
    - Constraint applies to days or overall plan
    - Constraint has rules that must be satisfied when generating plans
    
    Example Constraints:
    - "No fish on weekdays" (Dietary + Day restriction)
    - "Chicken max once per week" (Ingredient limit)
    - "Cooking time under 60 minutes" (Cooking time filter)
    """
    
    # Constraint category
    category = models.CharField(
        max_length=50,
        choices=[
            ('dietary', 'Dietary restriction (vegan, no dairy)'),
            ('ingredient_limit', 'Ingredient frequency limit'),
            ('cooking_time', 'Maximum cooking time'),
            ('allergy', 'Food allergy'),
            ('preference', 'Food preference/dislike'),
        ],
        help_text="Category of constraint"
    )
    
    # Constraint description (e.g., "No fish on weekdays")
    description = models.CharField(
        max_length=200,
        help_text="Human-readable constraint description"
    )
    
    # Days this constraint applies to (empty string or comma-separated)
    applicable_days = models.CharField(
        max_length=10,
        choices=[
            ('daily', 'Every day'),
            ('weekdays', 'Weekdays only'),
            ('weekends', 'Weekends only'),
            ('all', 'All days (overrides above)'),
        ],
        default='daily',
        help_text="Days this constraint applies to"
    )
    
    # For ingredient limit constraints: e.g., "chicken"
    target_ingredient = models.CharField(
        max_length=200,
        blank=True,
        help_text="Ingredient name for frequency limits (e.g., 'Chicken')",
        db_index=True
    )
    
    # Maximum occurrences per plan (only for ingredient_limit category)
    max_occurrences = models.PositiveIntegerField(
        null=True,
        blank=True,
        help_text="Max times this ingredient can appear in a week (e.g., 2)",
        validators=[MinValueValidator(1)]
    )
    
    class Meta:
        verbose_name = 'User Constraint'
        verbose_name_plural = 'User Constraints'
        ordering = ['-created_at']
    
    def __str__(self):
        return f"{self.description} ({self.category})"

class MealPlanner:
    """
    Engine for generating meal plans that respect user constraints.
    
    Usage:
        from planner.views import MealPlanner
        planner = MealPlanner(request.user)
        
        # Generate plan with constraints
        plan_data = planner.generate_weekly_plan(
            days=7,
            structure='balanced'
        )
    """
    
    @staticmethod
    def generate_weekly_plan(user, start_date=None, days=7, constraints=None):
        """
        Generate a meal plan for the week respecting user constraints.
        
        Args:
            user: Authenticated CustomUser
            start_date: Start date (defaults to next Monday)
            days: Number of days (1-7)
            constraints: List of UserConstraint objects
        
        Returns:
            dict with generated meal plan data including daily dishes
        """
        from recipes.models import Recipe, RecipeIngredient
        
        # Get user's owned recipes and public ones
        user_recipes = Recipe.objects.filter(created_by=user) | Recipe.objects.filter(is_public=True)
        
        # Filter by constraints (simplified - full logic in views.py)
        filtered_recipes = MealPlanner._apply_constraints(user_recipes, constraints)
        
        # Sort for variety: favorite first, then rating, then creation date
        filtered_recipes = filtered_recipes.order_by('-favorite', '-rating', '-created_at')
        
        # Generate plan for each day
        meal_plan = []
        ingredient_usage = {}  # Track used ingredients
        
        if not start_date:
            from datetime import datetime, timedelta
            today = datetime.now().date()
            
            # Find next Monday
            days_until_monday = (7 - today.weekday()) % 7
            start_date = today + timedelta(days=days_until_monday)
        
        for day_offset in range(days):
            day_date = start_date + timedelta(days=day_offset)
            
            # Skip weekends if user has weekday-only constraints
            should_skip_weekend = any(
                c.applicable_days == 'weekdays' 
                for c in constraints
            )
            
            if should_skip_weekend and day_date.weekday() >= 5:
                continue
            
            # Randomly select a recipe (respecting constraints)
            random_recipe = filtered_recipes.order_by('?').first() or filtered_recipes.first()
            
            meal_plan.append({
                'date': day_date,
                'recipe': random_recipe.id if random_recipe else None,
                'ingredients': MealPlanner._get_flat_ingredients(random_recipe) if random_recipe else [],
            })
            
            # Track ingredient usage for limits
            for ing in meal_plan[-1]['ingredients']:
                key = f"{ing['name']} {ing['unit']}".lower()
                ingredient_usage[key] = ingredient_usage.get(key, 0) + 1
        
        return {'days': meal_plan, 'ingredient_summary': ingredient_usage}

            if hasattr(ingredient_obj, 'ingredient_ref'):
