```python
# api/permissions.py
from rest_framework import permissions
from django.contrib.auth.models import AnonymousUser


class IsAuthenticatedOrReadOnly(permissions.BasePermission):
    """
    Allow GET, HEAD, OPTIONS without authentication;
    require authentication for CREATE, UPDATE, DELETE.
    
    Use Case: Read public data without login, write operations require auth.
    """
    
    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS:
            return True
        return bool(request.user and not isinstance(request.user, AnonymousUser))


class IsAuthenticatedOwnerOrReadOnly(permissions.BasePermission):
    """
    Allow read-only access to all objects;
    require authentication for CREATE, UPDATE, DELETE on own objects only.
    
    Use Case: Users can list/edit their own resources but read others'.
    """
    
    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS:
            return True
        return bool(request.user and not isinstance(request.user, AnonymousUser))
    
    def has_object_permission(self, request, view, obj):
        if request.method in permissions.SAFE_METHODS:
            return True
        
        # Check ownership or multi-tenancy filter
        from coremodels.models import BaseModel
        
        owner = getattr(obj, 'created_by', None)
        is_owner = owner and owner.id == request.user.id
        
        # Allow if owned by user, OR public flag set, OR user has explicit permission
        return is_owner or hasattr(obj, 'is_public') and obj.is_public


class HasExplicitPermission(permissions.BasePermission):
    """
    Check if user has explicit read permission via Permission model.
    
    Use Case: When resource is marked private but shared with specific users.
    """
    
    def has_object_permission(self, request, view, obj):
        # For read operations, check if user owns OR has read permission
        if request.method in permissions.SAFE_METHODS:
            from coremodels.models import BaseModel
            
            owner = getattr(obj, 'created_by', None)
            is_owner = owner and owner.id == request.user.id
            is_public = hasattr(obj, 'is_public') and obj.is_public
            has_explicit_permission = self._has_read_permission(request.user, obj)
            
            return is_owner or is_public or has_explicit_permission
        else:
            # For writes, must own the object
            from coremodels.models import BaseModel
            owner = getattr(obj, 'created_by', None)
            return bool(owner and owner.id == request.user.id)
    
    def _has_read_permission(self, user, obj):
        """Check if user has explicit read permission via Permission model."""
        from users.models import CustomUser, Permission
        
        if not user.is_authenticated:
            return False
        
        # Check shared permissions (read_only=True means can view)
        perms = Permission.objects.filter(
            object_type=type(obj).__name__,
            object_id=obj.id,
            granted_to=user,
            read_only=True
        )
        return perms.exists()


class UserHasIngredientPermission(permissions.BasePermission):
    """
    Special permission for ingredient models where nested recipe visibility
    depends on parent recipe's permissions.
    
    Use Case: Ingredient with nested_recipe shows only if the recipe is accessible.
    """
    
    def has_object_permission(self, request, view, obj):
        # For read operations, apply complex filtering logic
        if request.method in permissions.SAFE_METHODS:
            from coremodels.models import BaseModel
            
            owner = getattr(obj, 'created_by', None)
            is_owner = owner and owner.id == request.user.id
            
            # Check nested recipe visibility
            if hasattr(obj, 'nested_recipe'):
                nested_recipe = obj.nested_recipe
                if nested_recipe:
                    # Can only read nested ingredient if nested recipe is visible
                    nested_owner = getattr(nested_recipe, 'created_by', None)
                    nested_visible = (
                        nested_owner and nested_owner.id == request.user.id or
                        hasattr(nested_recipe, 'is_public') and nested_recipe.is_public
                    )
                    if not nested_visible:
                        # Check for explicit permission to read this ingredient
                        from users.models import CustomUser, Permission
                        has_permission = Permission.objects.filter(
                            object_type='ingredient',
                            object_id=obj.id,
                            granted_to=request.user,
                            read_only=True
                        ).exists()
                        return has_permission
            
            return is_owner or bool(hasattr(obj, 'is_public') and obj.is_public)
        else:
            return False


# API permission chain - apply to views in this order:
# 1. IsAuthenticatedOrReadOnly (basic read/write separation)
# 2. UserHasIngredientPermission (ingredient-specific nested filtering)
# 3. HasExplicitPermission (fallback for other models)
```

```python
from rest_framework import viewsets, filters, pagination
from rest_framework.response import Response
from coremodels.models import BaseModel

from ingredients.models import Ingredient
from recipes.models import Recipe
from dishes.models import Dish


class IngredientViewSet(viewsets.ReadOnlyModelViewSet):
    """
    Read-only ViewSet for public ingredient API.
    
    Endpoints: /api/v1/ingredients/
    
    Multi-tenancy: Shows ingredients where owner=user OR is_public=True
                  (or user has explicit permission to view)
    """
    
    queryset = Ingredient.objects.all()
    serializer_class = 'ingredients.serializers.IngredientSerializer'
    filter_backends = [filters.SearchFilter, filters.OrderingFilter]
    search_fields = ['name', 'unit']
    ordering_fields = ['name', 'created_at', 'updated_at']
    
    # Permission chain
    permission_classes = [
        IsAuthenticatedOrReadOnly,
        IngredientPermissionChecker,  # Custom checker for ingredient rules
    ]
    
    def get_queryset(self):
        """Apply multi-tenancy filter to queryset."""
        user = self.request.user
        
        if not user.is_authenticated:
            return Ingredient.objects.filter(
                is_public=True  # Only public ingredients for anonymous users
            )
        
        from ingredients.models import IngredientQuerySet, IngredientPermission
        
        # Use custom queryset with multi-tenancy filtering
        qs = Ingredient.objects.all()
        filtered_qs = IngredientQuerySet(qs).for_user(user)
        
        return filtered_qs
    
    def retrieve(self, request, *args, **kwargs):
        """Add permission check for single ingredient retrieval."""
        obj = self.get_object()
        if not self.check_object_permissions(request, obj):
            return Response({'detail': 'Not found.'}, status=404)
        return super().retrieve(request, *args, **kwargs)


class RecipeViewSet(viewsets.ReadOnlyModelViewSet):
    """
    Read-only ViewSet for public recipe API.
    
    Endpoints: /api/v1/recipes/
    
    Multi-tenancy: Shows recipes where owner=user OR is_public=True
                  (with permission-based filtering applied in serializer)
    """
    
    queryset = Recipe.objects.all()
    serializer_class = 'recipes.serializers.RecipeSerializer'
    filter_backends = [filters.SearchFilter, filters.OrderingFilter]
    search_fields = ['name', 'image_url']
    ordering_fields = ['name', 'rating', 'created_at']
    
    # Permission chain
    permission_classes = [
        IsAuthenticatedOrReadOnly,
    ]
    
    def get_queryset(self):
        """Apply multi-tenancy filter to queryset."""
        user = self.request.user
        
        if not user.is_authenticated:
            return Recipe.objects.filter(
                is_public=True  # Only public recipes for anonymous users
            )
        
        qs = Recipe.objects.all()
        filtered_qs = qs.filter(
            is_public=True  # Public recipes visible to all
        ) | qs.filter(
            created_by=user  # Owned by authenticated user
        )
        
        return filtered_qs
    
    def retrieve(self, request, *args, **kwargs):
        """Add permission check for single recipe retrieval."""
        obj = self.get_object()
        if not self.check_object_permissions(request, obj):
            return Response({'detail': 'Not found.'}, status=404)
        return super().retrieve(request, *args, **kwargs)

class IngredientPermissionChecker:
    """
    Custom permission checker for ingredient viewsets.
    Handles nested recipe visibility rules.
    """
    
    def has_object_permission(self, request, view, obj):
        if request.method in ['GET', 'HEAD', 'OPTIONS']:
            from coremodels.models import BaseModel
            
            owner = getattr(obj, 'created_by', None)
            is_owner = owner and owner.id == request.user.id
            
            # Check nested recipe visibility
            if hasattr(obj, 'parent_recipe') and obj.parent_recipe:
                nested_recipe = obj.parent_recipe
                if nested_recipe:
                    # Can only read nested ingredient if nested recipe is visible
                    nested_owner = getattr(nested_recipe, 'created_by', None)
                    nested_visible = (
                        nested_owner and nested_owner.id == request.user.id or
                        hasattr(nested_recipe, 'is_public') and nested_recipe.is_public
                    )
                    
                    if not nested_visible:
                        # Check for explicit permission to read this ingredient
                        from users.models import CustomUser, Permission
                        has_permission = Permission.objects.filter(
                            object_type='ingredient',
                            object_id=obj.id,
                            granted_to=request.user,
                            read_only=True
                        ).exists()
                        
                        return is_owner or has_permission
            
            return is_owner or bool(hasattr(obj, 'is_public') and obj.is_public)
        else:
            return False
```
